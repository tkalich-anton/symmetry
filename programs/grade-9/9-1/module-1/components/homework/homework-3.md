## Системы уравнений и неравенств.

$\left\{
  \begin{array}{l}
 2x - 1 > 3 \\
 3x - 2 < 11  \\
  \end{array}
\right.$

$\left\{
  \begin{array}{l}
    5x > 5x - 3 \\
 -3x < 2  \\
  \end{array}
\right.$

$\left\{
  \begin{array}{l}
    5(x+1) \leq 3(x+3) + 1 \\
     \dfrac{2x - 1}{7} \leq \dfrac{x + 1}{2} \\
  \end{array}
\right.$

$\left\{
  \begin{array}{l}
    3x + 4 \leq 4x + 6 \\
     x - 5 \leq 4 -2x \\
  \end{array}
\right.$

\right.$

    5x + 3 \leq 6x + 7 \\

  \end{array}
\right.$

  \end{array}
\right.$

$\left\{
