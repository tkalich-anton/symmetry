## Геометрия четырехугольника.

1) $\left\{
  \begin{array}{l}
     (x+6y)^2=7y, \\\\
     (x+6y)^2=7x; \\
  \end{array}
\right.$

3) $\left\{
  \begin{array}{l}
     3x^2+y=4, \\\\
     2x^2-y=1; \\
  \end{array}
\right.$

4) $\left\{
  \begin{array}{l}
     5x-4y=9, \\\\
     (5x-4y)^2=x^2+80; \\
  \end{array}
\right.$

5) $\left\{
  \begin{array}{l}
     2x+3y=7, \\\\
     2x^2+y=9; \\
  \end{array}
\right.$

6) $\left\{
  \begin{array}{l}
     (x-2)(y+3)=0, \\\\
     y^2+3y+1=x; \\
  \end{array}
\right.$

2) $\left\{
  \begin{array}{l}
     (3x-8y)^2=-5x, \\\\
     (3x-8y)^2=-5y; \\
  \end{array}
\right.$

3) $\left\{
  \begin{array}{l}
     8x+9y=6, \\\\
     (8x+9y)^2=x^2+27; \\
  \end{array}
\right.$

4) $\left\{
  \begin{array}{l}
     (x+6y)^2=7y, \\\\
     (x+6y)^2=7x; \\
  \end{array}
\right.$

5) $\left\{
  \begin{array}{l}
     3x+5y=-8, \\\\
     3x^2+y=2; \\
  \end{array}
\right.$

  \begin{array}{l}

  \end{array}
\right.$
