\begin{ex}
	\begin{condition}
		На катете \( BC  \) прямоугольного треугольника \( ABC \) как на диаметре построена окружность, которая пересекает гипотенузу \( AB  \) в точке \( K \). Найдите \( CK \), если \( BC = a  \) и \( AC = b \).
	\end{condition}
	\answer{\( \dfrac{ab}{\sqrt{a^2+b^2}} \)}
\end{ex}
