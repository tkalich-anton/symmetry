\begin{ex}
	\begin{condition}
		Найдите все значения параметра \( a \), при каждом из которых решением системы неравенств
		\[ \left\{
		\begin{array}{l}
			a+3x\le12,\\
			a+4x\ge x^2,\\
			a\le x
		\end{array}
		\right. \]
		является отрезок длины 2.
	\end{condition}
	\answer{\( -3; \dfrac{3}{2} \)}
\end{ex}
