\begin{ex}
	\begin{condition}
		Найдите все значения параметра \( a \), при каждом из которых система неравенств
		\[ \left\{
		\begin{array}{l}
			|x|+|a|\le4,\vspace{2mm}\\
			x^2+8x<16a+48
		\end{array}
		\right. \]
		имеет хотя бы одно решение на отрезке \( [-1;0] \).
	\end{condition}
	\answer{\( (8-8\sqrt{2};4) \)}
\end{ex}
