\begin{ex}
	\begin{condition}
		Найдите все значения \( a \), при каждом из которых система неравенств
		\[ \left\{
		\begin{array}{l}
			ax\ge2,\\
			\sqrt{x-1}>a,\\
			3x\le2a+11
		\end{array}
		\right. \]
		имеет хотя бы одно решение на отрезке \( [3; 4] \).
	\end{condition}
	\answer{\( \left[ \dfrac{1}{2};\sqrt{3} \right) \)}
\end{ex}
