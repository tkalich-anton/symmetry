		Найдите все значения параметра \( a \), для каждого из которых существует единственная тройка \( (x;y;z) \) действительных чисел \( x,y,z \), удовлетворяющая системе
		\[ \left\{
		\begin{array}{l}
			3^x+3^{\frac{1}{x}}=(a^2-1)+y^2+6,\\
			|y|z^4+2z^2-4a^2z+a+3=0.
		\end{array}
		\right. \]
	\end{condition}
	\answer{\( -1 \)}
