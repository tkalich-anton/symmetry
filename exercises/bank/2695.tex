\begin{ex}
	\begin{condition}
		В трапеции \( ABCD  \) меньшая диагональ \( BD  \) перпендикулярна основаниям \( AD  \) и \( BC \), сумма острых углов \( A  \) и \( C \) равна \( 90\degree \). Основания \( AD = a \), \( BC = b \). Найдите боковые стороны \( AB  \) и \( CD \).
	\end{condition}
	\answer{\( \sqrt{a(a+b)}, \sqrt{b(a+b)} \)}
\end{ex}
