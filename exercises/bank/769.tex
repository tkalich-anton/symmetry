\begin{ex}
	\begin{condition}
		Найдите все значения параметра \( a \), для каждого из которых система уравнений
		\[ \left\{
		\begin{array}{l}
			2^{\ln y}=4^{|x|},\\
			\log_2(x^4y^2+2a^2)=\log_2(1-ax^2y^2)+1.
		\end{array}
		\right. \] имеет единственное решение. При найденных значениях \( a \) решите систему.
	\end{condition}
	\answer{\( a=1, (0;1) \)}
\end{ex}
