		\begin{enumcols}
			\item Докажите, что плоскость \( \alpha \) параллельна прямой \( SA \).
			\item Найдите, в каком отношении плоскость \( \alpha \) делит объем пирамиды.
		\end{enumcols}
	\end{condition}
