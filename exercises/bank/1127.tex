\begin{ex}
	\begin{condition}
		Найдите все значения параметра \( a \), при каждом из которых существует единственная пара чисел \( (x;y) \), удовлетворяющая системе неравенств
		\[ \left\{
		\begin{array}{l}
			(x^2-xy+y^2)(x^2-36)\ge0,\\
			|x-2+y|+|x-2-y|\le a.
		\end{array}
		\right. \]
	\end{condition}
	\answer{\( a \in [4;8) \)}
\end{ex}
