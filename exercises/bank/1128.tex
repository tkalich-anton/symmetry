			\begin{enumcols}[label=\asbuk*)]
				\item Докажите, что прямые \( BM \) и \( MN \) перпендикулярны.
				\item Найдите угол между плоскостями \( BMN \) и \( ABB_1 \).
			\end{enumcols}
	\end{condition}
