\begin{ex}
	\begin{condition}
		Найдите все значения параметра \( a \), при каждом из которых имеет единственное решение система
		\[ \left\{
		\begin{array}{l}
			x+2y+xy+1=0,\\
			axy+x-y+1,5=0.
		\end{array}
		\right. \]
	\end{condition}
	\answer{\( -3,5\pm2\sqrt{2} \); \( -0,5 \); \( 1 \)}
\end{ex}
