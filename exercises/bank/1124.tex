\begin{ex}
	\begin{condition}
		Найдите все значения параметра \( a \), при каждом из которых система
		\[ \left\{
		\begin{array}{l}
			(x-a)^2+y^2=64,\\
			(|x|-8)^2+(|y|-15)^2=289.
		\end{array}
		\right. \]
		имеет ровно два решения.
	\end{condition}
	\answer{\( a \in \{-28\}\cup(-24;-8]\cup[8;24)\cup\{28\} \)}
\end{ex}
