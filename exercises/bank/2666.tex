\begin{ex}
	\begin{condition}
		Из точки \( M  \) проведены касательные \( MA  \) и \( MB  \) к окружности с центром \( O  \) (\( A  \) и \( B \) – точки касания). Найдите радиус окружности, если \( \angle AMB = \alpha  \) и \( \angle AB = a \).
	\end{condition}
	\answer{\( \dfrac{a}{2\cos(\alpha/2)} \)}
\end{ex}
