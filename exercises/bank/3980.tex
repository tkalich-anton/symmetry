\begin{ex}[type=inequality]
	\begin{condition}
		$ \log_{x + 3} (6 + x - x^2) + \log_{\sqrt{6 + x - x^2}}(x  + 3)\leqslant3 .$
	\end{condition}
	\answer{$ -2<x<\dfrac{1 - \sqrt{21}}{2},\ -\sqrt{3}\leqslant x\leqslant-\dfrac{3}{2},\ -1\leqslant x\leqslant\sqrt{3},\ \dfrac{1 + \sqrt{21}}{2}<x<3  .$}
\end{ex}
