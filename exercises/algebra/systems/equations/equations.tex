\section{Системы линейных уравнений}
\begin{enumerate}
	\item \textit{(Никольский 7 кл., №696, №697)} Решить систему уравнений:
	\begin{multicols}{2}
		\begin{enumerate}[label=\asbuk*)]
			\item 
			$\left\{
			\begin{array}{l}
				x-y-2=-1,\\
				x+y-5=0
			\end{array}
			\right.$ \answer{$(3;2)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x-y=2,\\
				x+y=6
			\end{array}
			\right.$ \answer{$(4;2)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x-2y=0,\\
				2x-3y-7=0\\
			\end{array}
			\right.$ \answer{$(14;7)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x+5y=0,\\
				3x+7y-16=0
			\end{array}
			\right.$ \answer{$(10;-2)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				y-3x=0,\\
				x-2y=-10
			\end{array}
			\right.$ \answer{$(2;6)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				7x-y=0,\\
				3x-y+12=-10
			\end{array}
			\right.$ \answer{$(3;21)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x-2y=3,\\
				5x+y=4
			\end{array}
			\right.$ \answer{$(1;-1)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x-y=2,\\
				3x-2y=9
			\end{array}
			\right.$ \answer{$(5;3)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x+2y-11=0,\\
				4x-5y=-8
			\end{array}
			\right.$ \answer{$(3;4)$}
			
			\item
			$\left\{
			\begin{array}{l}
				x+4y-2=0,\\
				3x+8y=2
			\end{array}
			\right.$ \answer{$(-2;1)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				2x+4y-90=0,\\
				x-3y=10
			\end{array}
			\right.$ \answer{$(31;7)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x-y-12=0,\\
				2x+4y=0
			\end{array}
			\right.$ \answer{$(8;-4)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				3x-2y-4=0,\\
				2x+10y-14=0
			\end{array}
			\right.$ \answer{$(2;1)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				3x-4y-7=0,\\
				x+2y+1=0
			\end{array}
			\right.$ \answer{$(1;-1)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				7x-2y+3=9,\\
				x+4y+7=-5
			\end{array}
			\right.$ \answer{$(0;-3)$}
		\end{enumerate}
	\end{multicols}
	\item \textit{(Никольский 7 кл., №698, №700)} Решить систему уравнений:
	\begin{multicols}{2}
		\begin{enumerate}[label=\asbuk*)]
			\item 
			$\left\{
			\begin{array}{l}
				5x+y-7=0,\\
				x-3y-11=0
			\end{array}
			\right.$ \answer{$(2;-3)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				2x+y-1=0,\\
				3x+2y+5=0
			\end{array}
			\right.$ \answer{$(7;-13)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				2x+y=7,\\
				x-2y+4=0
			\end{array}
			\right.$ \answer{$(2;3)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				3x+y=-5,\\
				x-3y-5=0
			\end{array}
			\right.$ \answer{$(-1;-2)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x+2y-4=0,\\
				3x+y+3=0
			\end{array}
			\right.$ \answer{$(-1;-2)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				5x+y-15=0,\\
				x-2y=14
			\end{array}
			\right.$ \answer{$(4;-5)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x+2y-3=0,\\
				x+y=-1
			\end{array}
			\right.$ \answer{$(-5;4)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x-3y+3=0,\\
				x+y=1
			\end{array}
			\right.$ \answer{$(0;1)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				4x+y-2=0,\\
				3x+y=-3
			\end{array}
			\right.$ \answer{$(5;-18)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x-y-7=0,\\
				3x-y+7=6
			\end{array}
			\right.$ \answer{$(-4;-11)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				2x-3y+7=0,\\
				3x+4y=1
			\end{array}
			\right.$ \answer{$\left(-\frac{25}{17};-\frac{23}{17}\right)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				3x-3y-5=0,\\
				6x+8y=-11
			\end{array}
			\right.$ \answer{$\left(\frac{1}{6};-\frac{3}{2}\right)$}
		\end{enumerate}
	\end{multicols}
	\item \textit{(Никольский 7 кл., №723-725)} Решить систему уравнений:
	\begin{multicols}{2}
		\begin{enumerate}[label=\asbuk*)]
			\item 
			$\left\{
			\begin{array}{l}
				x-y=5,\\
				-4x+4y=20
			\end{array}
			\right.$ \answer{Нет решений}
			
			\item 
			$\left\{
			\begin{array}{l}
				2x+3y=-4,\\
				5x-7=-6y
			\end{array}
			\right.$ \answer{$(15;-11\frac13)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				3x-2y=11,\\
				4x-5y=3
			\end{array}
			\right.$ \answer{$(7;5)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				5x+6y=13,\\
				7x+18y+1=0
			\end{array}
			\right.$ \answer{$(7;5)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				7x+6y=1,5,\\
				4x-9y-5=0
			\end{array}
			\right.$ \answer{$(\dfrac{1}{2};\dfrac{1}{3})$}
			
			\item 
			$\left\{
			\begin{array}{l}
				3x+4y=3,5,\\
				-3x-4y=40
			\end{array}
			\right.$ \answer{Нет решений}
			
			\item 
			$\left\{
			\begin{array}{l}
				\dfrac{x-3}{2}+\dfrac{y+4}{6}=2,\\
				\dfrac{1}{3}(x+2)-y=\dfrac{1}{3}
			\end{array}
			\right.$ \answer{$(5;2)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				\dfrac{5x}{2}+\dfrac{y}{5}+4=0,\\
				\dfrac{x}{3}+\dfrac{y}{6}=\dfrac{1}{6}
			\end{array}
			\right.$ \answer{$(-2;5)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				\dfrac{x+3}{2}-\dfrac{y-2}{3}=2,\\
				\dfrac{x-1}{4}+\dfrac{y+1}{3}=4
			\end{array}
			\right.$ \answer{$(5;8)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				\dfrac{x+y}{9}-\dfrac{x-y}{3}=2,\\
				\dfrac{2x-y}{6}-\dfrac{3x+2y}{3}=-20
			\end{array}
			\right.$ \answer{$(15;12)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				\dfrac{2x}{9}+\dfrac{y}{4}=0,\\
				\dfrac{5x}{12}+\dfrac{y}{3}=1
			\end{array}
			\right.$ \answer{$(\dfrac{108}{13});-\dfrac{96}{13}$}
			
			\item 
			$\left\{
			\begin{array}{l}
				\dfrac{x+y}{2}-\dfrac{2y}{3}=2\dfrac{1}{2},\\
				\dfrac{3x}{2}+2y=0
			\end{array}
			\right.$ \answer{$(4;-3)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				\dfrac{x+y}{2}-\dfrac{x-y}{3}=8,\\
				\dfrac{x+3}{3}+\dfrac{x-y}{4}=11
			\end{array}
			\right.$ \answer{$(\dfrac{372}{19};\dfrac{108}{19})$}
			
			\item 
			$\left\{
			\begin{array}{l}
				\dfrac{2x-1}{5}-\dfrac{3y-2}{4}=2,\\
				\dfrac{3x+1}{5}-\dfrac{3y+2}{4}=0
			\end{array}
			\right.$ \answer{$(3;2)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				\dfrac{2x-1}{5}+\dfrac{3y-2}{4}=2,\\
				\dfrac{3x+1}{5}-\dfrac{3y+2}{4}=0
			\end{array}
			\right.$ \answer{$(3;2)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				x+5=5+3x,\\
				x-3=9x+1
			\end{array}
			\right.$ \answer{Нет решений}
			
			\item 
			$\left\{
			\begin{array}{l}
				y+3=2y-4,\\
				2x+3=x
			\end{array}
			\right.$ \answer{$(-3;7)$}
			
			\item 
			$\left\{
			\begin{array}{l}
				3y-4=2-3y,\\
				y=1\dfrac{1}{3}-3y
			\end{array}
			\right.$ \answer{Нет решений}
			
			\item 
			$\left\{
			\begin{array}{l}
				x+y=x+y,\\
				x-y+2=0
			\end{array}
			\right.$ \answer{$(x; x+2)$, где $x$ – любое число}
			
			\item 
			$\left\{
			\begin{array}{l}
				2x+3y=2x+3y+2,\\
				x-7y+1=0
			\end{array}
			\right.$ \answer{Нет решений}
			
			\item 
			$\left\{
			\begin{array}{l}
				3x+5y=5(x+3y)-2(x+5y),\\
				y-3+x=2x+(x+y-3)
			\end{array}
			\right.$ \answer{$(0;y)$, где $y$ – любое число}
			
			\item 
			$\left\{
			\begin{array}{l}
				3x+4y+1=(x+y-2)+(2x+3y+3),\\
				x+y+2=y+(2+x)
			\end{array}
			\right.$ \answer{$(x;y)$, где $x,y$ – любые число}
			
			
		\end{enumerate}
	\end{multicols}
\end{enumerate}
